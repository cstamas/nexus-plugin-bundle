27  /**
28   * This is Hg "id" command but with "--debug" switch, to force Hg show full checksum, not just the part of it as it
29   * usually does.
30   *
31   * @since 1.0
32   */
33  public class HgDebugIdCommand
34      extends AbstractCommand
35  {
36      @Override
37      protected HgDebugIdScmResult executeCommand(final ScmProviderRepository repository,
38                                                  final ScmFileSet fileSet,
39                                                  final CommandParameters parameters)
40          throws ScmException
41      {
42          HgOutputConsumer consumer;
43          ScmResult result;
44  
45          consumer = new HgOutputConsumer(getLogger());
46          result = HgUtils.execute(consumer, getLogger(), fileSet.getBasedir(), new String[]{"id", "-i", "--debug"});
47          checkResult(result);
48  
49          String changeSetHash = consumer.getOutput();
50  
51          // trim off the possible "+"
52          if (changeSetHash.endsWith("+")) {
53              changeSetHash = changeSetHash.substring(0, changeSetHash.length() - 1);
54          }
55  
56          consumer = new HgOutputConsumer(getLogger());
57          result = HgUtils.execute(consumer, getLogger(), fileSet.getBasedir(), new String[]{
58              "log",
59              "-r",
60              String.valueOf(changeSetHash),
61              "--template",
62              "\"{date|isodate}\""
63          });
64          checkResult(result);
65  
66          return new HgDebugIdScmResult(null, null, null, true, changeSetHash, consumer.getOutput());
67      }
68  
69      private void checkResult(final ScmResult result)
70          throws ScmException
71      {
72          if (!result.isSuccess()) {
73              getLogger().debug("Provider message:");
74              getLogger().debug(result.getProviderMessage() == null ? "" : result.getProviderMessage());
75              getLogger().debug("Command output:");
76              getLogger().debug(result.getCommandOutput() == null ? "" : result.getCommandOutput());
77              throw new ScmException("Command failed: " + StringUtils.defaultString(result.getProviderMessage()));
78          }
79      }
80  
81      private static class HgOutputConsumer
82          extends HgConsumer
83      {
84          private String output;
85  
86          private HgOutputConsumer(final ScmLogger logger) {
87              super(logger);
88          }
89  
90          public void doConsume(final ScmFileStatus status, final String line) {
91              output = line;
92          }
93  
94          private String getOutput() {
95              return output;
96          }
97      }
98  }
